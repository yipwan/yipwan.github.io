            curr = GetNext(curr, high, ref flags);
        } while (curr != start && dsq[curr] > epsSqr);  // 当前点的距离大于阈值，持续向后找
        if (curr == start) break;
    }

    int prev = GetPrior(curr, high, ref flags);   
    int next = GetNext(curr, high, ref flags);
    if (next == prev) break;

    int prior2;
    if (dsq[next] < dsq[curr]) // 这里对相邻点做了比较，选择相邻较小的垂直举例的点进行移除。个人觉得用处不大。
    {
        prior2 = prev;
        prev = curr;
        curr = next; // 当前点比下一个点更小，跳过当前点不保留。
        next = GetNext(next, high, ref flags);
    }
    else
        prior2 = GetPrior(prev, high, ref flags);

    flags[curr] = true;
    curr = next;
    next = GetNext(next, high, ref flags);
    if (isClosedPath || ((curr != high) && (curr != 0)))
        dsq[curr] = PerpendicDistFromLineSqrd(path[curr], path[prev], path[next]);
    if (isClosedPath || ((prev != 0) && (prev != high)))
        dsq[prev] = PerpendicDistFromLineSqrd(path[prev], path[prior2], path[curr]);
}

    裁剪数据

    1. struct Scanline 含义：表示扫描线的信息。扫描线是裁剪算法中用于处理多边形交点和边的水平线。 作用：在扫描线算法中，Scanline 用于存储与当前扫描线相关的边和交点信息，帮助算法追踪多边形的轮廓。
    2. struct IntersectNode 含义：表示两个边的交点信息。 作用：在多边形裁剪过程中，IntersectNode 用于存储边与边的交点，这些交点对于计算多边形的交集、并集等布尔运算至关重要。
    3. struct Active 含义：表示当前活动的边。 作用：在裁剪算法中，Active 结构用于管理当前扫描线上的活动边列表（Active Edge List, AEL）。这些活动边在扫描线移动过程中不断更新，用于追踪多边形的轮廓。
    4. struct Vertex 含义：表示多边形的一个顶点。 作用：Vertex 用于存储多边形的顶点信息，包括顶点的坐标和与之相关的边信息。它是多边形表示的基础单元。
    5. struct LocalMinima 含义：表示局部最小点。 作用：在多边形处理中，LocalMinima 用于识别多边形轮廓上的局部最小点，这些点对于确定多边形的轮廓和处理凹多边形非常重要。
    6. struct OutRec 含义：表示输出记录。 作用：OutRec 用于存储裁剪结果的相关信息，包括裁剪后的多边形轮廓和与之相关的边。它是裁剪算法输出的关键数据结构。
    7. struct HorzSegment 含义：表示水平线段。 作用：在处理多边形时，HorzSegment 用于存储水平线段的信息，这些线段对于处理多边形的水平边和交点非常有用。 这些数据结构共同构成了 Clipper 库的核心，使得它能够高效地执行多边形裁剪和布尔运算。

    Active-活跃边

    在 Clipper 库中，Active 结构是用于管理多边形裁剪过程中活动边的关键数据结构。以下是对 Active 结构中每个属性的详细解释： 1. Point64 bot 含义：表示边的底部点（即边的起点）。 作用：在裁剪过程中，bot 用于确定边的起始位置。 2. Point64 top 含义：表示边的顶部点（即边的终点）。 作用：在裁剪过程中，top 用于确定边的结束位置。 3. int64_t curr_x = 0 含义：表示当前扫描线上的 x 坐标。 作用：在扫描线移动过程中，curr_x 会不断更新，用于追踪边在当前扫描线上的位置。 4. double dx = 0.0 含义：表示边的斜率（即 x 坐标的变化率）。 作用：在扫描线移动过程中，dx 用于计算边在下一个扫描线上的 x 坐标。 5. int wind_dx = 1 含义：表示边的绕向（顺时针或逆时针）。 作用：在裁剪过程中，wind_dx 用于确定边的方向，值为 1 或 -1。 6. int wind_cnt = 0 含义：表示边的绕数（winding count）。 作用：在裁剪过程中，wind_cnt 用于追踪边的绕数，用于处理多边形的填充规则。 7. int wind_cnt2 = 0 含义：表示边的绕数（winding count）的相反多边形类型。 作用：在裁剪过程中，wind_cnt2 用于处理多边形的布尔运算，例如并集、交集等。 8. OutRec outrec = nullptr 含义：指向与当前边相关的输出记录。 作用：在裁剪过程中，outrec 用于存储裁剪结果的相关信息。 9. Active prev_in_ael = nullptr 含义：指向当前边在活动边表（AEL）中的前一个边。 作用：在扫描线移动过程中，prev_in_ael 用于维护活动边表的链表结构。 10. Active next_in_ael = nullptr 含义：指向当前边在活动边表（AEL）中的下一个边。 作用：在扫描线移动过程中，next_in_ael 用于维护活动边表的链表结构。 11. Active prev_in_sel = nullptr 含义：指向当前边在排序边表（SEL）中的前一个边。 作用：在排序边表中，prev_in_sel 用于维护排序边表的链表结构。 12. Active next_in_sel = nullptr 含义：指向当前边在排序边表（SEL）中的下一个边。 作用：在排序边表中，next_in_sel 用于维护排序边表的链表结构。 13. Active jump = nullptr 含义：指向当前边的跳转边。 作用：在处理水平边时，jump 用于跳过某些边，优化处理过程。 14. Vertex vertex_top = nullptr 含义：指向边的顶部顶点。 作用：在处理多边形顶点时，vertex_top 用于存储边的顶部顶点信息。 15. LocalMinima local_min = nullptr 含义：指向边的局部最小点。 作用：在处理多边形的局部最小点时，local_min 用于存储边的局部最小点信息。 16. bool is_left_bound = false 含义：表示当前边是否是左侧边界。 作用：在处理多边形边界时，is_left_bound 用于区分左侧和右侧边界。 17. JoinWith join_with = JoinWith::NoJoin 含义：表示边的连接方式。 作用：在处理边的连接时，join_with 用于指定边的连接方式，例如是否需要连接

    ps：关于localminima的解释如下。每个封闭路径（即多边形）由一系列顶点组成，这些顶点形成交替上升和下降的边“边界”。所谓“局部最小值”是指上升边界和下降边界在底部相交的顶点，而“局部最大值”是指上升边界和下降边界在顶部相交的顶点。

    在多边形裁剪和布尔运算中，水平边（即与 x 轴平行的边）需要特殊处理，因为它们不会像普通斜边那样与扫描线相交。DoHorizontal 函数的作用通常包括以下几点：

    1. 处理水平边的交点：
    2. 水平边不会与扫描线产生交点，因此需要单独处理它们与其他非水平边的交点。
    3. 该函数可能会检查水平边与其他边的相交情况，并更新交点信息。
    4. 更新水平边的连接关系：
    5. 在多边形布尔运算中，水平边可能会影响多边形的拓扑结构。DoHorizontal 函数可能会处理水平边与其他边的连接关系，确保多边形的边界正确。
    6. 处理水平边的输出：
    7. 在生成最终的裁剪结果时，水平边需要被正确地包含在输出多边形中。该函数可能会负责将水平边的顶点加入到输出多边形的路径中。
